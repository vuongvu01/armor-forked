## Basic usage

<Preview>
    <Story name="Basic usage" height="450px">
        {() => {
            const structure = useMemo(() => ([
                {
                    label: 'Birds',
                    id: 'birds',
                    expanded: true,
                    selected: true,
                    items: [
                        {
                            label: 'Hawk',
                            id: 'hawk',
                            selected: true,
                        },
                        {
                            label: 'Sparrowhawk',
                            id: 'sparrowhawk',
                        },
                    ],
                },
                {
                    label: 'Bugs',
                    id: 'bugs',
                    expanded: false,
                    items: [
                        {
                            label: 'Dragonfly',
                            id: 'dragonfly',
                        },
                        {
                            label: 'Glowworm',
                            id: 'glowworm',
                        },
                        {
                            label: 'Ladybug',
                            id: 'ladybug',
                        },
                    ],
                },
            ]), []);
            
            const [ items, setItems ] = useState(structure);
            
            const onElementClick = (itemId, item) => {
                item.expanded = !item.expanded;
                
                setItems([...items]);
            };
            
            return (<Navigation items={structure} onElementClick={onElementClick} />);
        }}
    </Story>
</Preview>
