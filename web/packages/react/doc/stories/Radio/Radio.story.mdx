## Radio Group

<Preview>
    <Story name="Radio group" height="190px">
        {() => {
             const groupName = 'radioGroup';
             const [groupSelectedValue, setSelectedValueGroup] = useState('');
             const handleChange = event => {
                 const selectedValue = event.target.value;
                 if (selectedValue) {
                     setSelectedValueGroup(selectedValue);
                 }
             };
             return (
        <>
        <RadioGroup
             name={groupName}
             selectedValue={groupSelectedValue}
             onChange={handleChange}
        >
            <Radio value="pizza" label="Pizza" />
            <Radio value="pasta" label="Pasta" />
            <Radio value="risotto" label="Risotto" />
        </RadioGroup>
    </>
             );
         }}
    </Story>
</Preview>
