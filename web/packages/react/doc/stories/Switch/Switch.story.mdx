export const MyDynamicSwitch = () => {
  const [isChecked, setChecked] = useState(true);
  const [isDisabled, setDisabled] = useState(false);
  return (
    <>
        <Button onClick={() => setChecked(!isChecked)}>Toggle</Button>
        <Button marginLeft={2} onClick={() => setDisabled(!isDisabled)} >Disable/Enable</Button>
        <br />
        <br />
        <Switch checked={isChecked} disabled={isDisabled} label="External control" />
    </>
  )
}
