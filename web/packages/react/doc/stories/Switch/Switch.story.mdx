  const [isChecked, setChecked] = useState(true);
  return (
    <>
        <h4>External control</h4>
        <Button onClick={() => setChecked(!isChecked)}> Toggle checkbox state </Button>
        <Switch checked={isChecked} onChange={() => {}}  />
    </>
  )
}
