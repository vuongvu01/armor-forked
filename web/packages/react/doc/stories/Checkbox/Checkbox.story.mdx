export const MyDynamicCheckbox = () => {
  const [isChecked, setChecked] = useState(false);
  return (
    <>
        <Button onClick={() => setChecked(!isChecked)}> Toggle checkbox state </Button>
        <Checkbox checked={isChecked} onChange={() => {}}  />
    </>
  )
}
